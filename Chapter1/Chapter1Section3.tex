\textbf{\large{The Product Rule}} \par

\textbf{Ex 1.3.1: } $\diff \left[x^2\sin (x)\right]$ \\[11pt]
\textbf{Sol 1.3.1: } \begin{align*}
    \diff \left[x^2\sin (x)\right] & = x^2 \cdot \cos (x) + \sin (x) \cdot (2x) \\[11pt]
    & = \boxed{x^2\cos (x) + 2(x)\sin (x)}
\end{align*}
